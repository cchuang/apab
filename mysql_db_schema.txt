CREATE DATABASE `apab` /*!40100 DEFAULT CHARACTER SET latin1 */;

CREATE TABLE `audience_resp` (
  `idaudience_resp` int(11) NOT NULL AUTO_INCREMENT,
  `val` varchar(45) DEFAULT NULL,
  `time` timestamp NOT NULL DEFAULT CURRENT_TIMESTAMP,
  `agent` varchar(100) DEFAULT NULL,
  `slideno` int(11) DEFAULT NULL,
  `ip` varchar(45) DEFAULT NULL,
  `handle` varchar(45) DEFAULT NULL,
  PRIMARY KEY (`idaudience_resp`)
) ENGINE=InnoDB DEFAULT CHARSET=latin1;

CREATE TABLE `status` (
  `idstatus` int(11) NOT NULL AUTO_INCREMENT,
  `slideno` int(11) NOT NULL,
  `path` varchar(45) NOT NULL,
  `time` timestamp NOT NULL DEFAULT CURRENT_TIMESTAMP,
  `handle` varchar(45) NOT NULL DEFAULT 'assessment.php',
  PRIMARY KEY (`idstatus`)
) ENGINE=InnoDB DEFAULT CHARSET=latin1 COMMENT='current status';
